PowerExchange 9.5.0 配置文件参考

1:启用Oracle数据库归档 

ALTER SYSTEM SET log_archive_dest_1 = 'location=/u01/oracle/Desktop/archivelog' SCOPE=SPFILE;

SHUTDOWN IMMEDIATE;

STARTUP MOUNT;

ALTER DATABASE ARCHIVELOG;

ALTER DATABASE OPEN;

SHUTDOWN IMMEDIATE;

STARTUP;
