Greenplum导出导出表结构表数据

导出表结构
pg_dump -E UTF8 -s -v DBname -n bi_dim > /1.ddl

导出数据
pg_dump -E UTF8 -a -v DBname -n bi_dim > /1.data

导入
psql -U gpadmin -f /1.ddl DBname

/* 导入导出,切换用户 -a 数据，-s 模式 */
-- 导出schema
pg_dump -U 数据库用户名 -s -n 模式名1 -n 模式名2 -n 模式名3 -n 模式名4 -n 模式名5 -n 模式名6 -f 导出的文件名 数据库名;
pg_dump -U 数据库用户名 -s -n 模式名 -f 导出的文件名 数据库名;
-- 导出数据
pg_dump -U 数据库用户名 -a -n 模式名1 -n 模式名2 -n 模式名3 -f 导出的文件名 数据库名;
pg_dump -U 数据库用户名 -a --column-inserts -n 模式名1 -n 模式名2 -f 导出的文件名 数据库名;
-- 导出表
pg_dump -U 数据库用户名 -a -n 模式名 -t 表明 -f 导出的文件名 数据库名;
pg_dump -U 数据库用户名 -ad -n 模式名 -t 表名 -f 导出的文件名 数据库名;
pg_dump -U 数据库用户名 -a --column-inserts -n 模式名 -t 表名 -f 导出的文件名 数据库名;
-- 导入SQL
psql -U 数据库用户名 -f 导入的文件名 dbname
